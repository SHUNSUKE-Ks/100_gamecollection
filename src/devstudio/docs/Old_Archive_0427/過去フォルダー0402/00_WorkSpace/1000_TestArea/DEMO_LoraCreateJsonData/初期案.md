<script>

let DB = {}

async function loadData(){
  const res = await fetch("assets/tags.json")
  DB = await res.json()
  render()
}

function render(){
  renderList()
  renderGallery()
}

function renderList(){
  const el = document.getElementById("listView")
  el.innerHTML=""

  Object.entries(DB).forEach(([file,tags])=>{
    const div=document.createElement("div")
    div.className="list-item"

    div.innerHTML=`
      <b>${file}</b><br>
      ${tags.map(t=>`<span class="tag">${t}</span>`).join("")}
      <br>
      <input placeholder="add tag" onkeydown="addTag(event,'${file}')">
    `
    el.appendChild(div)
  })
}

function renderGallery(){
  const el=document.getElementById("galleryView")
  el.innerHTML=`<div class="gallery"></div>`
  const grid=el.firstChild

  Object.entries(DB).forEach(([file,tags])=>{
    const card=document.createElement("div")
    card.className="card"

    card.innerHTML=`
      <img src="assets/images/${file}">
      <div><b>${file}</b></div>
      <div>${tags.map(t=>`<span class="tag">${t}</span>`).join("")}</div>
    `
    grid.appendChild(card)
  })
}

function addTag(e,file){
  if(e.key!=="Enter") return

  const val=e.target.value.trim()
  if(!val) return

  DB[file].push(val)
  e.target.value=""
  render()
}

function switchView(v){
  document.querySelectorAll(".view").forEach(el=>el.classList.remove("active"))
  document.getElementById(v+"View").classList.add("active")
}

loadData()

</script>
